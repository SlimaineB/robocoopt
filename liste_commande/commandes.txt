#First, download the Laravel installer using Composer:
composer global require laravel/installer

#Create project 
laravel new robocoopt

#Create missions model
php artisan make:migration create_missions_table --create=missions

#Update C:\wamp64\www\robocoopt\app\Providers\AppServiceProvider.php avec Schema::defaultStringLength(191);

#Update database/migrations/* script and Create database missions model :
php artisan migrate

#Create MissionController
php artisan make:controller MissionController --resource --model=Mission

#To feed database with fake data: create factories/MissionFactory and run :
php artisan tinker
factory(App\Mission::class, 200)->create();


#Install advanced datatables 
composer require yajra/laravel-datatables-oracle
